    # Quick Start

    # cf-ui-win

    基于 Vue2 的 Windows 风格 Web 组件库

    # 安装

    npm install cf-ui-win

    # 使用说明

    // 全部引用
import CfUiWin from "cf-ui-win/lib/index/index.js";
import "cf-ui-win/lib/index/index.css";
Vue.use(CfUiWin);

// 按需引用
import CfButton from "cf-ui-win/lib/button/index.js";
import "cf-ui-win/lib/button/index.css";
Vue.use(CfButton);
